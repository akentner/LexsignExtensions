  103. 103                 }
  104. 104             }
  105. 105         }
  106. 106 
  107. 107         foreach ($fromSchema->getSequences() AS $sequenceName => $sequence) {
  108. 108             if (!$toSchema->hasSequence($sequenceName)) {
  109. 109                 $diff->removedSequences[] = $sequence;

  233. 233         foreach ($toFkeys AS $key2 => $constraint2) {
