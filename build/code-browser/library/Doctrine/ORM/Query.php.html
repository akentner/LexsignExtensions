  234. 234         foreach ($this->_params as $key => $value) {
  235. 235             if ( ! isset($paramMappings[$key])) {
  236. 236                 throw QueryException::unknownParameter($key);
  237. 237             }

  240. 240                     $types[$position] = $this->_paramTypes[$key];
  241. 241                 }
  242. 242             }

  244. 244             if (is_object($value) && $this->_em->getMetadataFactory()->hasMetadataFor(get_class($value))) {
  245. 245                 if ($this->_em->getUnitOfWork()->getEntityState($value) == UnitOfWork::STATE_MANAGED) {
  246. 246                     $idValues = $this->_em->getUnitOfWork()->getEntityIdentifier($value);
  247. 247                 } else {
  248. 248                     $class = $this->_em->getClassMetadata(get_class($value));
  249. 249                     $idValues = $class->getIdentifierValues($value);
  250. 250                 }
  251. 251                 $sqlPositions = $paramMappings[$key];
  252. 252                 $cSqlPos = count($sqlPositions);
  253. 253                 $cIdValues = count($idValues);
  254. 254                 $idValues = array_values($idValues);
  255. 255                 for ($i = 0; $i < $cSqlPos; $i++) {
  256. 256                     $sqlParams[$sqlPositions[$i]] = $idValues[ ($i % $cIdValues) ];
  257. 257                 }
  258. 258             } else {
  259. 259                 foreach ($paramMappings[$key] as $position) {
  260. 260                     $sqlParams[$position] = $value;
  261. 261                 }
  262. 262             }
  263. 263         }
  264. 264 
  265. 265         if ($sqlParams) {
  266. 266             ksort($sqlParams);
  267. 267             $sqlParams = array_values($sqlParams);
  268. 268         }
  269. 269 
  270. 270         if ($this->_resultSetMapping === null) {
  271. 271             $this->_resultSetMapping = $this->_parserResult->getResultSetMapping();
  272. 272         }
  273. 273 
  274. 274         return $executor->execute($this->_em->getConnection(), $sqlParams, $types);
