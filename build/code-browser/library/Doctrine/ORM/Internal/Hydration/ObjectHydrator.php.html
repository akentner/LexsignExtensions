  241. 241     /**
  242. 242      * Hydrates a single row in an SQL result set.
  243. 243      * 
  244. 244      * @internal
  245. 245      * First, the data of the row is split into chunks where each chunk contains data
  246. 246      * that belongs to a particular component/class. Afterwards, all these chunks
  247. 247      * are processed, one after the other. For each chunk of class data only one of the
  248. 248      * following code paths is executed:
  249. 249      * 
  250. 250      * Path A: The data chunk belongs to a joined/associated object and the association
  251. 251      *         is collection-valued.
  252. 252      * Path B: The data chunk belongs to a joined/associated object and the association
  253. 253      *         is single-valued.
  254. 254      * Path C: The data chunk belongs to a root result element/object that appears in the topmost
  255. 255      *         level of the hydrated result. A typical example are the objects of the type
  256. 256      *         specified by the FROM clause in a DQL query. 
  257. 257      * 
  258. 258      * @param array $data The data of the row to process.
  259. 259      * @param array $cache The cache to use.
  260. 260      * @param array $result The result array to fill.
  261. 261      */
  262. 262     protected function _hydrateRow(array $data, array &$cache, array &$result)
  263. 263     {
  264. 264         // Initialize
  265. 265         $id = $this->_idTemplate; // initialize the id-memory
  266. 266         $nonemptyComponents = array();
  267. 267         // Split the row data into chunks of class data.
  268. 268         $rowData = $this->_gatherRowData($data, $cache, $id, $nonemptyComponents);
  269. 269 
  270. 270         // Extract scalar values. They're appended at the end.
  271. 271         if (isset($rowData['scalars'])) {
  272. 272             $scalars = $rowData['scalars'];
  273. 273             unset($rowData['scalars']);
  274. 274             if (empty($rowData)) {
  275. 275                 ++$this->_resultCounter;
  276. 276             }
  277. 277         }
  278. 278 
  279. 279         // Hydrate the data chunks
  280. 280         foreach ($rowData as $dqlAlias => $data) {
  281. 281             $entityName = $this->_rsm->aliasMap[$dqlAlias];
  282. 282 
  283. 283             if (isset($this->_rsm->parentAliasMap[$dqlAlias])) {
  284. 284                 // It's a joined result
  285. 285 
  286. 286                 $parentAlias = $this->_rsm->parentAliasMap[$dqlAlias];
  287. 287                 // we need the $path to save into the identifier map which entities were already
  288. 288                 // seen for this parent-child relationship
  289. 289                 $path = $parentAlias . '.' . $dqlAlias;
  290. 290 
  291. 291                 // Get a reference to the parent object to which the joined element belongs.
  292. 292                 if ($this->_rsm->isMixed && isset($this->_rootAliases[$parentAlias])) {
  293. 293                     $first = reset($this->_resultPointers);
  294. 294                     $parentObject = $this->_resultPointers[$parentAlias][key($first)];
  295. 295                 } else if (isset($this->_resultPointers[$parentAlias])) {
  296. 296                     $parentObject = $this->_resultPointers[$parentAlias];
  297. 297                 } else {
  298. 298                     // Parent object of relation not found, so skip it.
  299. 299                     continue;
  300. 300                 }
  301. 301 
  302. 302                 $parentClass = $this->_ce[$this->_rsm->aliasMap[$parentAlias]];
  303. 303                 $oid = spl_object_hash($parentObject);
  304. 304                 $relationField = $this->_rsm->relationMap[$dqlAlias];
  305. 305                 $relation = $parentClass->associationMappings[$relationField];
  306. 306                 $reflField = $parentClass->reflFields[$relationField];
  307. 307 
  308. 308                 // Check the type of the relation (many or single-valued)
  309. 309                 if ( ! ($relation['type'] & ClassMetadata::TO_ONE)) {
  310. 310                     // PATH A: Collection-valued association
  311. 311                     if (isset($nonemptyComponents[$dqlAlias])) {
  312. 312                         $collKey = $oid . $relationField;
  313. 313                         if (isset($this->_initializedCollections[$collKey])) {
  314. 314                             $reflFieldValue = $this->_initializedCollections[$collKey];
  315. 315                         } else if ( ! isset($this->_existingCollections[$collKey])) {
  316. 316                             $reflFieldValue = $this->_initRelatedCollection($parentObject, $parentClass, $relationField);
  317. 317                         }
  318. 318 
  319. 319                         $indexExists = isset($this->_identifierMap[$path][$id[$parentAlias]][$id[$dqlAlias]]);
  320. 320                         $index = $indexExists ? $this->_identifierMap[$path][$id[$parentAlias]][$id[$dqlAlias]] : false;
  321. 321                         $indexIsValid = $index !== false ? isset($reflFieldValue[$index]) : false;
  322. 322 
  323. 323                         if ( ! $indexExists || ! $indexIsValid) {
  324. 324                             if (isset($this->_existingCollections[$collKey])) {
  325. 325                                 // Collection exists, only look for the element in the identity map.
  326. 326                                 if ($element = $this->_getEntityFromIdentityMap($entityName, $data)) {
  327. 327                                     $this->_resultPointers[$dqlAlias] = $element;
  328. 328                                 } else {
  329. 329                                     unset($this->_resultPointers[$dqlAlias]);
  330. 330                                 }
  331. 331                             } else {
  332. 332                                 $element = $this->_getEntity($data, $dqlAlias);
  333. 333 
  334. 334                                 if (isset($this->_rsm->indexByMap[$dqlAlias])) {
  335. 335                                     $field = $this->_rsm->indexByMap[$dqlAlias];
  336. 336                                     $indexValue = $this->_ce[$entityName]->reflFields[$field]->getValue($element);
  337. 337                                     $reflFieldValue->hydrateSet($indexValue, $element);
  338. 338                                     $this->_identifierMap[$path][$id[$parentAlias]][$id[$dqlAlias]] = $indexValue;
  339. 339                                 } else {
  340. 340                                     $reflFieldValue->hydrateAdd($element);
  341. 341                                     $reflFieldValue->last();
  342. 342                                     $this->_identifierMap[$path][$id[$parentAlias]][$id[$dqlAlias]] = $reflFieldValue->key();
  343. 343                                 }
  344. 344                                 // Update result pointer
  345. 345                                 $this->_resultPointers[$dqlAlias] = $element;
  346. 346                             }
  347. 347                         } else {
  348. 348                             // Update result pointer
  349. 349                             $this->_resultPointers[$dqlAlias] = $reflFieldValue[$index];
  350. 350                         }
  351. 351                     } else if ( ! $reflField->getValue($parentObject)) {
  352. 352                         $coll = new PersistentCollection($this->_em, $this->_ce[$entityName], new ArrayCollection);
  353. 353                         $coll->setOwner($parentObject, $relation);
  354. 354                         $reflField->setValue($parentObject, $coll);
  355. 355                         $this->_uow->setOriginalEntityProperty($oid, $relationField, $coll);
  356. 356                     }
  357. 357                 } else {
  358. 358                     // PATH B: Single-valued association
  359. 359                     $reflFieldValue = $reflField->getValue($parentObject);
  360. 360                     if ( ! $reflFieldValue || isset($this->_hints[Query::HINT_REFRESH])) {
  361. 361                         if (isset($nonemptyComponents[$dqlAlias])) {
  362. 362                             $element = $this->_getEntity($data, $dqlAlias);
  363. 363                             $reflField->setValue($parentObject, $element);
  364. 364                             $this->_uow->setOriginalEntityProperty($oid, $relationField, $element);
  365. 365                             $targetClass = $this->_ce[$relation['targetEntity']];
  366. 366                             if ($relation['isOwningSide']) {
  367. 367                                 //TODO: Just check hints['fetched'] here?
  368. 368                                 // If there is an inverse mapping on the target class its bidirectional
  369. 369                                 if ($relation['inversedBy']) {
  370. 370                                     $inverseAssoc = $targetClass->associationMappings[$relation['inversedBy']];
  371. 371                                     if ($inverseAssoc['type'] & ClassMetadata::TO_ONE) {
  372. 372                                         $targetClass->reflFields[$inverseAssoc['fieldName']]->setValue($element, $parentObject);
  373. 373                                         $this->_uow->setOriginalEntityProperty(spl_object_hash($element), $inverseAssoc['fieldName'], $parentObject);
  374. 374                                     }
  375. 375                                 } else if ($parentClass === $targetClass && $relation['mappedBy']) {
  376. 376                                     // Special case: bi-directional self-referencing one-one on the same class
  377. 377                                     $targetClass->reflFields[$relationField]->setValue($element, $parentObject);
