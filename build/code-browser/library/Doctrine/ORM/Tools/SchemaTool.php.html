  429. 429     /**

  431. 431      *
  432. 432      * Previously this was a simple task, but with DDC-117 this problem is actually recursive. If its
  433. 433      * not a simple field, go through all identifier field names that are associations recursivly and
  434. 434      * find that referenced column name.
  435. 435      *
  436. 436      * TODO: Is there any way to make this code more pleasing?
  437. 437      *
  438. 438      * @param ClassMetadata $class
  439. 439      * @param string $referencedColumnName
  440. 440      * @return array(ClassMetadata, referencedFieldName)
  441. 441      */
  442. 442     private function getDefiningClass($class, $referencedColumnName)
  443. 443     {

  445. 445 
  446. 446         if ($class->hasField($referencedFieldName)) {
  447. 447             return array($class, $referencedFieldName);
  448. 448         } else if (in_array($referencedColumnName, $class->getIdentifierColumnNames())) {

  450. 450             foreach ($class->getIdentifierFieldNames() AS $fieldName) {

  453. 453                         $this->_em->getClassMetadata($class->associationMappings[$fieldName]['targetEntity']),
  454. 454                         $class->getSingleAssociationReferencedJoinColumnName($fieldName)
  455. 455                     );
  456. 456                 }
  457. 457             }
  458. 458         }
  459. 459 
  460. 460         return null;
  461. 461     }

  463. 463     /**
  464. 464      * Gather columns and fk constraints that are required for one part of relationship.
  465. 465      *
  466. 466      * @param array $joinColumns
  467. 467      * @param \Doctrine\DBAL\Schema\Table $theJoinTable
  468. 468      * @param ClassMetadata $class
  469. 469      * @param array $mapping
  470. 470      * @param array $primaryKeyColumns
  471. 471      * @param array $uniqueConstraints
  472. 472      */
  473. 473     private function _gatherRelationJoinColumns($joinColumns, $theJoinTable, $class, $mapping, &$primaryKeyColumns, &$uniqueConstraints)
  474. 474     {
  475. 475         $localColumns = array();
  476. 476         $foreignColumns = array();
  477. 477         $fkOptions = array();
  478. 478         $foreignTableName = $class->getQuotedTableName($this->_platform);
  479. 479 
  480. 480         foreach ($joinColumns as $joinColumn) {
  481. 481             $columnName = $joinColumn['name'];
  482. 482             list($definingClass, $referencedFieldName) = $this->getDefiningClass($class, $joinColumn['referencedColumnName']);
  483. 483 
  484. 484             if (!$definingClass) {
  485. 485                 throw new \Doctrine\ORM\ORMException(
  486. 486                     "Column name `".$joinColumn['referencedColumnName']."` referenced for relation from ".
  487. 487                     $mapping['sourceEntity'] . " towards ". $mapping['targetEntity'] . " does not exist."
  488. 488                 );
  489. 489             }
  490. 490 
  491. 491             $primaryKeyColumns[] = $columnName;
  492. 492             $localColumns[] = $columnName;
  493. 493             $foreignColumns[] = $joinColumn['referencedColumnName'];
  494. 494 
  495. 495             if ( ! $theJoinTable->hasColumn($joinColumn['name'])) {
  496. 496                 // Only add the column to the table if it does not exist already.
  497. 497                 // It might exist already if the foreign key is mapped into a regular
  498. 498                 // property as well.
  499. 499 
  500. 500                 $fieldMapping = $definingClass->getFieldMapping($referencedFieldName);
  501. 501 
  502. 502                 $columnDef = null;
  503. 503                 if (isset($joinColumn['columnDefinition'])) {
  504. 504                     $columnDef = $joinColumn['columnDefinition'];
  505. 505                 } else if (isset($fieldMapping['columnDefinition'])) {
  506. 506                     $columnDef = $fieldMapping['columnDefinition'];
  507. 507                 }
  508. 508                 $columnOptions = array('notnull' => false, 'columnDefinition' => $columnDef);
  509. 509                 if (isset($joinColumn['nullable'])) {
  510. 510                     $columnOptions['notnull'] = !$joinColumn['nullable'];
  511. 511                 }
  512. 512                 if ($fieldMapping['type'] == "string" && isset($fieldMapping['length'])) {
  513. 513                     $columnOptions['length'] = $fieldMapping['length'];
  514. 514                 } else if ($fieldMapping['type'] == "decimal") {
  515. 515                     $columnOptions['scale'] = $fieldMapping['scale'];
  516. 516                     $columnOptions['precision'] = $fieldMapping['precision'];
  517. 517                 }
  518. 518 
  519. 519                 $theJoinTable->addColumn($columnName, $fieldMapping['type'], $columnOptions);
  520. 520             }
  521. 521 
  522. 522             if (isset($joinColumn['unique']) && $joinColumn['unique'] == true) {
  523. 523                 $uniqueConstraints[] = array('columns' => array($columnName));
  524. 524             }
  525. 525 
  526. 526             if (isset($joinColumn['onUpdate'])) {
  527. 527                 $fkOptions['onUpdate'] = $joinColumn['onUpdate'];
  528. 528             }
  529. 529 
  530. 530             if (isset($joinColumn['onDelete'])) {
  531. 531                 $fkOptions['onDelete'] = $joinColumn['onDelete'];
  532. 532             }
  533. 533         }
  534. 534 
  535. 535         $theJoinTable->addUnnamedForeignKeyConstraint(
  536. 536             $foreignTableName, $localColumns, $foreignColumns, $fkOptions
  537. 537         );
  538. 538     }
