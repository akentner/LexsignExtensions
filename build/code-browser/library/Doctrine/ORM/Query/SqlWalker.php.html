  256.  256                   . ' ' . $tableAlias . ' ON ';

  262.  262                 $sql .= $baseTableAlias . '.' . $columnName

  375.  375             $sql .= ' ORDER BY '.$orderBySql;
  376.  376         }

  378.  378 
  379.  379         $sql = $this->_platform->modifyLimitQuery(
  380.  380             $sql, $this->_query->getMaxResults(), $this->_query->getFirstResult()
  381.  381         );
  382.  382 
  383.  383         if (($lockMode = $this->_query->getHint(Query::HINT_LOCK_MODE)) !== false) {
  384.  384             if ($lockMode == LockMode::PESSIMISTIC_READ) {
  385.  385                 $sql .= " " . $this->_platform->getReadLockSQL();
  386.  386             } else if ($lockMode == LockMode::PESSIMISTIC_WRITE) {
  387.  387                 $sql .= " " . $this->_platform->getWriteLockSQL();
  388.  388             } else if ($lockMode == LockMode::OPTIMISTIC) {
  389.  389                 foreach ($this->_selectedClasses AS $class) {
  390.  390                     if ( ! $class->isVersioned) {
  391.  391                         throw \Doctrine\ORM\OptimisticLockException::lockFailed();
  392.  392                     }
  393.  393                 }
  394.  394             }
  395.  395         }
  396.  396
