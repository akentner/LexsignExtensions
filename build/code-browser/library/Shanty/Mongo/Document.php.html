  536.  536     /**
  537.  537      * Get a property
  538.  538      * 
  539.  539      * @param mixed $property
  540.  540      */
  541.  541     public function getProperty($property)
  542.  542     {
  543.  543         // If property exists and initialised then return it
  544.  544         if (array_key_exists($property, $this->_data)) {
  545.  545             return $this->_data[$property];
  546.  546         }
  547.  547         
  548.  548         // Fetch clean data for this property
  549.  549         if (array_key_exists($property, $this->_cleanData)) {
  550.  550             $data = $this->_cleanData[$property];
  551.  551         }
  552.  552         else {
  553.  553             $data = array();
  554.  554         }
  555.  555         
  556.  556         // If data is not an array then we can do nothing else with it
  557.  557         if (!is_array($data)) {
  558.  558             $this->_data[$property] = $data;
  559.  559             return $this->_data[$property];
  560.  560         }
  561.  561     
  562.  562         // If property is supposed to be an array then initialise an array
  563.  563         if ($this->hasRequirement($property, 'Array')) {
  564.  564             return $this->_data[$property] = $data;
  565.  565         }
  566.  566         
  567.  567         // If property is a reference to another document then fetch the reference document

  572.  572

  574.  574             if (!$data) {
  575.  575                 $this->_data[$property] = null;
  576.  576                 return $this->_data[$property];
  577.  577             }
  578.  578             
  579.  579             $reference = true;
  580.  580         }
  581.  581         else {
  582.  582             $collection = $this->getConfigAttribute('collection');
  583.  583             $reference = false;
  584.  584         }
  585.  585

  595.  595                 $className = 'Shanty_Mongo_Document';
  596.  596             }
  597.  597             
  598.  598             if ($className) $docType = 'Shanty_Mongo_Document';
  599.  599         }
  600.  600         
  601.  601         // Nothing else to do
  602.  602         if (!$className) return null;
  603.  603         
  604.  604         // Configure property for document/documentSet usage
  605.  605         $config = array();

  607.  607         $config['connectionGroup'] = $this->getConfigAttribute('connectionGroup');
  608.  608         $config['db'] = $this->getConfigAttribute('db');
  609.  609         $config['collection'] = $collection;
  610.  610         $config['requirementModifiers'] = $this->getRequirements($property.'.');
  611.  611         $config['hasId'] = $this->hasRequirement($property, 'hasId');
  612.  612         
  613.  613         if (!$reference) {
  614.  614             $config['pathToDocument'] = $this->getPathToProperty($property);
  615.  615             $config['criteria'] = $this->getCriteria();
  616.  616         }
  617.  617         
  618.  618         // Initialise document
  619.  619         $document = new $className($data, $config);
  620.  620         
  621.  621         // if this document was a reference then remember that
  622.  622         if ($reference) {
  623.  623             $this->_references->attach($document);
  624.  624         }
  625.  625         
  626.  626         $this->_data[$property] = $document;
  627.  627         return $this->_data[$property];
  628.  628     }
  629.  629     
  630.  630     /**
  631.  631      * Set a property
  632.  632      * 
  633.  633      * @param mixed $property
  634.  634      * @param mixed $value
  635.  635      */
  636.  636     public function setProperty($property, $value)
  637.  637     {
  638.  638         if (substr($property, 0, 1) == '_') {
  639.  639             require_once 'Shanty/Mongo/Exception.php';
  640.  640             throw new Shanty_Mongo_Exception("Can not set private property '$property'");
  641.  641         }
  642.  642         
  643.  643         $validators = $this->getValidators($property);
  644.  644         
  645.  645         // Throw exception if value is not valid
  646.  646         if (!is_null($value) && !$validators->isValid($value)) {
  647.  647             require_once 'Shanty/Mongo/Exception.php';
  648.  648             throw new Shanty_Mongo_Exception(implode($validators->getMessages(), "\n"));
  649.  649         }
  650.  650         
  651.  651         // Unset property
  652.  652         if (is_null($value)) {
  653.  653             $this->_data[$property] = null;
  654.  654             return;
  655.  655         }
  656.  656         
  657.  657         if ($value instanceof Shanty_Mongo_Document && !$this->hasRequirement($property, 'AsReference')) {
  658.  658             if (!$value->isNewDocument() || !$value->isRootDocument()) {
  659.  659                 $documentClass = get_class($value);
  660.  660                 $value = new $documentClass($value->export(), array('new' => false, 'pathToDocument' => $this->getPathToProperty($property)));
  661.  661             }
  662.  662             else {
  663.  663                 $value->setPathToDocument($this->getPathToProperty($property));
  664.  664             }
  665.  665             
  666.  666             $value->setConfigAttribute('connectionGroup', $this->getConfigAttribute('connectionGroup'));
  667.  667             $value->setConfigAttribute('db', $this->getConfigAttribute('db'));
  668.  668             $value->setConfigAttribute('collection', $this->getConfigAttribute('collection'));
  669.  669             $value->setConfigAttribute('criteria', $this->getCriteria());
  670.  670             $value->applyRequirements($this->getRequirements($property.'.'));
  671.  671         }
  672.  672         
  673.  673         // Filter value
  674.  674         $value = $this->getFilters($property)->filter($value);
  675.  675         
  676.  676         $this->_data[$property] = $value;
  677.  677     }
  678.  678     
  679.  679     /**
  680.  680      * Determine if this document has a property

  690.  690         }
  691.  691         
  692.  692         // If property has not been initialised
  693.  693         if (array_key_exists($property, $this->_cleanData)) {
  694.  694             return !is_null($this->_cleanData[$property]);
  695.  695         }
  696.  696         
  697.  697         return false;
  698.  698     }

  758.  758     /**

  761.  761      * @return array
  762.  762      */
  763.  763     public function export()
  764.  764     {
  765.  765         $exportData = $this->_cleanData;
  766.  766         
  767.  767         foreach ($this->_data as $property => $value) {
  768.  768             // If property has been deleted
  769.  769             if (is_null($value)) {
  770.  770                 unset($exportData[$property]);
  771.  771                 continue;
  772.  772             }
  773.  773             
  774.  774             // If property is a document

  779.  779                 }

  782.  782                 }

  786.  786                     $exportData[$property] = $value->createReference();
  787.  787                     continue;
  788.  788                 }
  789.  789                 
  790.  790                 $data = $value->export();

  792.  792                     $exportData[$property] = $data;
  793.  793                 }

  795.  795             }
  796.  796

  798.  798         }
  799.  799         
  800.  800         // make sure required properties are not empty

  802.  802         foreach ($requiredProperties as $property) {

  804.  804                 require_once 'Shanty/Mongo/Exception.php';
  805.  805                 throw new Shanty_Mongo_Exception("Property '{$property}' must not be null.");
  806.  806             }
  807.  807         }
  808.  808         
  809.  809         return $exportData;
  810.  810     }

  826.  826      */
  827.  827     public function isEmpty()
  828.  828     {
  829.  829         $doNoCount = array();
  830.  830         
  831.  831         foreach ($this->_data as $property => $value) {
  832.  832             if ($value instanceof Shanty_Mongo_Document) {
  833.  833                 if (!$value->isEmpty()) return false;
  834.  834             }
  835.  835             elseif (!is_null($value)) {
  836.  836                 return false;
  837.  837             }
  838.  838             
  839.  839             $doNoCount[] = $property;
  840.  840         }
  841.  841         
  842.  842         foreach ($this->_cleanData as $property => $value) {

  844.  844                 continue;
  845.  845             }
  846.  846             
  847.  847             if (!is_null($value)) {
  848.  848                 return false;
  849.  849             }

  855.  855     /**
  856.  856      * Convert data changes into operations
  857.  857      * 
  858.  858      * @param array $data
  859.  859      */
  860.  860     public function processChanges(array $data = array())
  861.  861     {
  862.  862         foreach ($data as $property => $value) {
  863.  863             if ($property === '_id') continue;
  864.  864             
  865.  865             if (!array_key_exists($property, $this->_cleanData)) {
  866.  866                 $this->addOperation('$set', $property, $value);
  867.  867                 continue;
  868.  868             }
  869.  869             
  870.  870             $newValue = $value;
  871.  871             $oldValue = $this->_cleanData[$property];
  872.  872             
  873.  873             if (MongoDBRef::isRef($newValue) && MongoDBRef::isRef($oldValue)) {
  874.  874                 $newValue['$id'] = $newValue['$id']->__toString();
  875.  875                 $oldValue['$id'] = $oldValue['$id']->__toString();
  876.  876             }
  877.  877             
  878.  878             if ($newValue !== $oldValue) {
  879.  879                 $this->addOperation('$set', $property, $value);
  880.  880             }
  881.  881         }
  882.  882         
  883.  883         foreach ($this->_cleanData as $property => $value) {
  884.  884             if (array_key_exists($property, $data)) continue;
  885.  885             
  886.  886             $this->addOperation('$unset', $property, 1);
  887.  887         }
  888.  888     }

  890.  890     /**
